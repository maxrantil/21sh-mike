pid_t    process_group_set(pid_t pid, pid_t job_first_pid, bool foreground)
{
    pid_t    target;

    if (job_first_pid > 0)
        target = job_first_pid;
    else
        target = pid;
    if (setpgid(pid, target) == -1)
        return (-1);
    if (foreground)
    {
        if (ioctl(STDIN_FILENO, TIOCSPGRP, &target) == -1)
            return (-1);
    }
    return (target);
}
